while True:
    # 1. Estrai obiettivi da rigeneproject.org
    new_objectives = extract_from_rigene()
    
    # 2. Analizza current state
    current_state = assess_speace_state()
    
    # 3. Genera mutation proposals
    proposals = generate_proposals(new_objectives, current_state)
    
    # 4. Evaluate via fitness function

        fitness = evaluate_fitness(proposal)
